# Aliyun服务器备份

## MySQL备份

```mysql
create table b_blog
(
    id            int auto_increment
        primary key,
    name          varchar(40)                         not null,
    url           varchar(255)                        not null,
    content       varchar(16000)                      not null,
    readAmount    int       default 0                 not null,
    likeAmount    int       default 0                 not null,
    commentAmount int       default 0                 not null,
    uploadTime    timestamp default CURRENT_TIMESTAMP not null
);

create table b_blog_comment
(
    id        int auto_increment
        primary key,
    blogId    int not null,
    commentId int not null
);

create table b_blog_like
(
    id     int auto_increment
        primary key,
    blogID int null,
    likeId int null
);

create table b_blog_tag
(
    id     int auto_increment
        primary key,
    blogId int not null,
    tagId  int not null
);

create table b_comment
(
    id         int auto_increment
        primary key,
    content    varchar(255)                        null,
    uploadTime timestamp default CURRENT_TIMESTAMP not null
);

create table b_like
(
    id       int auto_increment
        primary key,
    blogId   int                                 null,
    userId   int                                 null,
    likeTime timestamp default CURRENT_TIMESTAMP null
);

create table b_tag
(
    id          int auto_increment
        primary key,
    name        varchar(40)  not null,
    description varchar(255) not null
);

create table t_perms
(
    id   int           not null
        primary key,
    name varbinary(80) null,
    url  varchar(255)  null
);

create table t_role
(
    id   int         not null
        primary key,
    name varchar(50) null
);

create table t_role_perms
(
    id      int not null
        primary key,
    roleid  int null,
    permsid int null
);

create table t_user
(
    id          int auto_increment
        primary key,
    username    varchar(40)  null,
    password    varchar(40)  null,
    salt        varchar(255) null,
    phone       int          null,
    email       varchar(40)  null,
    create_time timestamp    null
);

create table t_user_role
(
    id     int not null
        primary key,
    userid int null,
    roleid int null
);


```
